AIM: Write a prgram to implement DDA algorithm.

#include <graphics.h>
#include <math.h>
#include <iostream>

using namespace std;

int main()
{
	float x1, x2, y1, y2, dx, dy, l, i;
	cout << "Enter the value of x1, y1, x2 and y2\n";
	cin >> x1 >> y1 >> x2 >> y2;
	dx = x2 - x1;
	dy = y2 - y1;
	if(abs(dx) > abs(dy))
		l = abs(dx);
	else
		l = abs(dy);

	dy = dy / l;
	x1 += 0.5;
	y2 += 0.5;
	dx = dx / l;
	int gdriver = DETECT, gmode = VGAMAX;
	initgraph(&gdriver, &gmode, "NULL");
	
	for(i = 0; i <= l; i++)
	{
		putpixel(x1, y1, WHITE);
		x1 += dx;
		y1 += dy;
	}
	delay(1000);
	return 0;
}

OUTPUT: Various lines were successfully drawn.
